local player = game.Players.LocalPlayer
local char = player.Character or player.CharacterAdded:Wait()
local humanoid = char:WaitForChild("Humanoid")

local frozenTracks = {}
local locked = false

local function freezeAll()
	for _, track in pairs(humanoid:GetPlayingAnimationTracks()) do
		if track.IsPlaying then
			track:AdjustSpeed(0)
			table.insert(frozenTracks, track)
		end
	end
	local emoteController = humanoid:FindFirstChildOfClass("EmoteController")
	if emoteController then
		emoteController:Destroy()
	end
end

local function unfreezeAll()
	for _, track in pairs(frozenTracks) do
		if track and track.IsPlaying then
			track:AdjustSpeed(1)
		end
	end
	frozenTracks = {}
end

humanoid.AnimationPlayed:Connect(function(track)
	if locked then
		task.wait(0.05)
		if track.IsPlaying then
			track:AdjustSpeed(0)
			table.insert(frozenTracks, track)
		end
	end
end)

local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/bloodball/-back-ups-for-libs/main/youngstar"))()
local Window = library:CreateWindow("travar animações")

Window:Button("On (Travar Animações)", function()
	locked = true
	freezeAll()
end)

Window:Button("Off (Destravar Animações)", function()
	locked = false
	unfreezeAll()
end)
